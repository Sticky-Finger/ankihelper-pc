  // lib/theme/theme_provider.dart

  import 'package:flutter/material.dart';
  import 'package:flutter_riverpod/flutter_riverpod.dart';
  import 'fluent_tokens.dart';

  enum AppTheme { light, dark }

  class ThemeState {
    final AppTheme theme;
    final FluentTokens tokens;

    ThemeState(this.theme) : tokens = FluentTokens(isDark: theme ==
  AppTheme.dark);

    ThemeState copyWith({AppTheme? theme}) =>
        ThemeState(theme ?? this.theme);
  }

  class ThemeNotifier extends StateNotifier<ThemeState> {
    ThemeNotifier() : super(ThemeState(AppTheme.light));

    void toggleTheme() {
      state = state.theme == AppTheme.light
          ? ThemeState(AppTheme.dark)
          : ThemeState(AppTheme.light);
    }
  }

  final themeProvider = StateNotifierProvider<ThemeNotifier,
  ThemeState>((ref) {
    return ThemeNotifier();
  });

  // lib/widgets/title_bar.dart

  import 'package:flutter/material.dart';
  import 'package:flutter_riverpod/flutter_riverpod.dart';
  import '../theme/fluent_tokens.dart';
  import '../theme/theme_provider.dart';

  class TitleBar extends ConsumerWidget {
    const TitleBar({super.key});

    @override
    Widget build(BuildContext context, WidgetRef ref) {
      final tokens = ref.watch(themeProvider).tokens;

      return Container(
        height: 44,
        padding: const EdgeInsets.symmetric(horizontal: 16),
        decoration: BoxDecoration(
          color: tokens.bgCard,
          border: Border(
            bottom: BorderSide(color: tokens.stroke3, width: 1),
          ),
        ),
        child: Row(
          children: [
            // 左侧：图标 + 标题 + 版本号
            Row(
              children: [
                // ... SVG 图标
                const SizedBox(width: 8),
                Text('Anki划词助手', style: TextStyle(
                  fontSize: 14, fontWeight: FontWeight.w600, color:
  tokens.fg1)),
                const SizedBox(width: 2),
                Text('v0.1', style: TextStyle(
                  fontSize: 12, color: tokens.fg4)),
              ],
            ),
            const Spacer(),
            // 右侧：主题切换、设置、关于、手动输入按钮
            Row(
              children: [
                _IconButton(
                  icon: Icons.light_mode,
                  onPressed: () =>
  ref.read(themeProvider.notifier).toggleTheme(),
                  tokens: tokens,
                ),
                _TextButton(label: '设置', tokens: tokens),
                _TextButton(label: '关于', tokens: tokens),
                _TextButton(label: '手动输入', tokens: tokens),
              ],
            ),
          ],
        ),
      );
    }
  }
